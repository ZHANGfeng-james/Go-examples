~~~go
func (p *Point) ScaleBy(factor float64) {
    p.X *= factor
    p.Y *= factor
}

func main() {
    var p = Point{1, 1}
    var q = Point{5, 4}
    fmt.Println(p.Distance(q)) // 5
    
    p.ScaleBy(2) // {2 2}
	q.ScaleBy(2) // {10 8}

	fmt.Println(p.Distance(q)) // 10
}
~~~
